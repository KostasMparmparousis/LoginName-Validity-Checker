  "$schema": "http://json-schema.org/draft-07/schema",
  "type": "object",
  "properties": {

      "type": "integer",
      "examples": [
        "100"
      ]
    },
    "message": {
      "type": "string",
      "examples": [
        "No Conflicts Found"
      ]
    },
    "personPairedWith": {
      "type": "array",
      "items": {
        "type": "string"
      }
    },
    "suggestions": {
      "type": "array",
      "items": {
        "type": "string"
      }
    }
  }
}
